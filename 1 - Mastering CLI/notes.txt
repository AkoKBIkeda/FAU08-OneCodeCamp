# Make a folder (directory)
mkdir [directory_name]
mkdir test
mkdir "test 1"

# Remove a directory/folder
rmdir [directory_name]
rmdir test
rmdir "test 1"

# Print/present working (current) directory
pwd

# List contents of the folder
ls

# Change directory
cd [directory_path]
cd test

# Go back one directory
cd ..

# Create a file
touch [file_name.ext]
touch test.txt

# Remove a file
rm [file_name.ext]
rm test.txt
